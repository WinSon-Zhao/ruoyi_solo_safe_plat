### 2.3 审计日志哈希链

```java
@Component
public class AuditLogChain {
    
    @Inject
    private SecAuditLogMapper auditLogMapper;
    
    /**
     * 计算日志哈希
     */
    public String calculateHash(SecAuditLog log, String prevHash) {
        String data = String.format("%s|%s|%s|%s|%s|%s",
            log.getLogId(),
            log.getOperationType(),
            log.getRequestUrl(),
            log.getUserId(),
            log.getOperationTime(),
            prevHash
        );
        return DigestUtil.sha256Hex(data);
    }
    
    /**
     * 验证日志链完整性
     */
    public boolean verifyChain() {
        List<SecAuditLog> logs = auditLogMapper.selectAllOrderByTime();
        
        for (int i = 1; i < logs.size(); i++) {
            SecAuditLog current = logs.get(i);
            SecAuditLog previous = logs.get(i - 1);
            
            // 验证前一条哈希
            if (!current.getPrevHash().equals(previous.getCurrHash())) {
                return false;
            }
            
            // 验证当前哈希
            String expectedHash = calculateHash(current, current.getPrevHash());
            if (!expectedHash.equals(current.getCurrHash())) {
                return false;
            }
        }
        
        return true;
    }
}
```

### 3.1 国密算法支持

```java
@Component
public class SmCryptoService {
    
    /**
     * SM2 非对称加密/签名
     */
    public byte[] sm2Encrypt(byte[] data, PublicKey publicKey) {
        SM2Engine engine = new SM2Engine();
        engine.init(true, new ParametersWithRandom(
            new ECPublicKeyParameters(...), 
            new SecureRandom()
        ));
        return engine.processBlock(data, 0, data.length);
    }
    
    public byte[] sm2Sign(byte[] data, PrivateKey privateKey) {
        SM2Signer signer = new SM2Signer();
        signer.init(true, new ECKeyParameters(...));
        signer.update(data, 0, data.length);
        return signer.generateSignature();
    }
    
    /**
     * SM3 哈希
     */
    public byte[] sm3Hash(byte[] data) {
        SM3Digest digest = new SM3Digest();
        digest.update(data, 0, data.length);
        byte[] result = new byte[digest.getDigestSize()];
        digest.doFinal(result, 0);
        return result;
    }
    
    /**
     * SM4 对称加密
     */
    public byte[] sm4Encrypt(byte[] data, byte[] key) {
        SM4Engine engine = new SM4Engine();
        engine.init(true, new KeyParameter(key));
        
        PaddedBufferedBlockCipher cipher = new PaddedBufferedBlockCipher(
            new CBCBlockCipher(engine), new PKCS7Padding()
        );
        cipher.init(true, new ParametersWithIV(new KeyParameter(key), iv));
        
        byte[] output = new byte[cipher.getOutputSize(data.length)];
        int len = cipher.processBytes(data, 0, data.length, output, 0);
        cipher.doFinal(output, len);
        return output;
    }
}
```

### 4.2 数据脱敏工具

```java
@Component
public class DataMaskingUtil {
    
    /**
     * 手机号脱敏
     */
    public static String maskPhone(String phone) {
        if (StrUtil.isBlank(phone) || phone.length() != 11) {
            return phone;
        }
        return phone.substring(0, 3) + "****" + phone.substring(7);
    }
    
    /**
     * 邮箱脱敏
     */
    public static String maskEmail(String email) {
        if (StrUtil.isBlank(email) || !email.contains("@")) {
            return email;
        }
        String[] parts = email.split("@");
        String name = parts[0];
        String domain = parts[1];
        
        if (name.length() <= 2) {
            return "*@" + domain;
        }
        return name.substring(0, 2) + "***@" + domain;
    }
    
    /**
     * IP脱敏
     */
    public static String maskIp(String ip) {
        if (StrUtil.isBlank(ip)) {
            return ip;
        }
        // 192.168.1.100 -> 192.168.*.*
        String[] parts = ip.split("\\.");
        if (parts.length != 4) {
            return ip;
        }
        return parts[0] + "." + parts[1] + ".*.*";
    }
    
    /**
     * 通用脱敏
     */
    public static String mask(String value, String rule) {
        return switch (rule) {
            case "phone" -> maskPhone(value);
            case "email" -> maskEmail(value);
            case "ip" -> maskIp(value);
            case "id_card" -> maskIdCard(value);
            default -> value;
        };
    }
}
```
